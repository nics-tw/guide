{{- $current_content := .input_content | htmlEscape -}}
{{- with .highlight_terms_string -}}
  {{- $highlight_terms := split . "," -}}
  {{- range $highlight_terms -}}
    {{- $term_to_highlight := . -}}
    {{- $escaped_term := $term_to_highlight | htmlEscape -}}
    {{- $current_content = replace $current_content $escaped_term (printf "%s" $escaped_term) -}}
  {{- end -}}
  {{- $current_content | safeHTML -}}
{{- else -}}
  {{- $current_content | safeHTML -}}
{{- end -}}
